Because makeSquare needs to modify its receiver, be

package main

import "fmt"

type Rectangle struct {
	Length float64
	Width  float64
}

func rekt(rek Rectangle) {
	fmt.Println(rek.Length, "x", rek.Width)
}

func (re Rectangle) makeSquare() {
	if re.Length > re.Width {
		re.Length = re.Width
	} else {
		re.Width = re.Length
	}
}

func main() {
	var r Rectangle
	r.Length = 10
	r.Width = 11
	r.makeSquare()
	rekt(r)
	var re Rectangle
	re.Length = 9
	re.Width = 11
	re.makeSquare()
	rekt(re)
}
