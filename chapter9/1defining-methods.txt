package main

import "fmt"

type rectangle struct {
	length float64
	width  float64
}

// YOUR CODE HERE: Convert this function to a method on
// the "rectangle" type named "info".
func (info rectangle) rectangleInfo() {
	fmt.Println("Length:", info.length)
	fmt.Println("Width:", info.width)
}

func main() {
	var r rectangle
	r.length = 4.2
	r.width = 2.4
	r.rectangleInfo()
}
